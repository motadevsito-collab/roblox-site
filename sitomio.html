Esempio di codice Lua base:

-- Muovi un cubo in avanti
local cubo = workspace.Cubo
cubo.Position = cubo.Position + Vector3.new(1,0,0)

Spiegazione: workspace.Cubo è l’oggetto, Position indica la posizione, Vector3.new(1,0,0) significa "sposta di 1 passo sull'asse X".

💻 Script Lua Avanzati

Ecco alcuni script pronti da usare o modificare!

-- Script: Movimento semplice del personaggio
local player = game.Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoid = character:WaitForChild("Humanoid")

humanoid.WalkSpeed = 20  -- Aumenta velocità

-- Salto extra alto
humanoid.JumpPower = 100
humanoid.Jump = true

-- Teletrasporto
local part = workspace.TeleportPart
part.Touched:Connect(function(hit)
    local character = hit.Parent
    if character:FindFirstChild("Humanoid") then
        character:SetPrimaryPartCFrame(CFrame.new(50,10,50))
    end
end)

-- Mostra messaggio sullo schermo
local player = game.Players.LocalPlayer
local gui = Instance.new("ScreenGui", player.PlayerGui)
local label = Instance.new("TextLabel", gui)
label.Text = "Benvenuto nel gioco!"
label.Size = UDim2.new(0,300,0,50)
label.Position = UDim2.new(0.5,-150,0.1,0)
label.TextScaled = true
label.BackgroundColor3 = Color3.fromRGB(255,182,193)

-- Sistema punteggio
local leaderstats = Instance.new("Folder")
leaderstats.Name = "leaderstats"
leaderstats.Parent = game.Players.LocalPlayer

local score = Instance.new("IntValue")
score.Name = "Punti"
score.Value = 0
score.Parent = leaderstats

-- Raccogli oggetto
local part = workspace.Oggetto
part.Touched:Connect(function(hit)
    local character = hit.Parent
    if character:FindFirstChild("Humanoid") then
        part:Destroy()
        print("Oggetto raccolto!")
    end
end)

-- Porta che teletrasporta
local porta = workspace.Porta
porta.Touched:Connect(function(hit)
    if hit.Parent:FindFirstChild("Humanoid") then
        hit.Parent:SetPrimaryPartCFrame(CFrame.new(0,10,0))
    end
end)

-- Suono interattivo
local sound = Instance.new("Sound", workspace)
sound.SoundId = "rbxassetid://12345678"
workspace.Part.Touched:Connect(function(hit)
    sound:Play()
end)

-- Animazione semplice
local player = game.Players.LocalPlayer
local animator = player.Character.Humanoid:FindFirstChildOfClass("Animator")
local anim = Instance.new("Animation")
anim.AnimationId = "rbxassetid://87654321"
animator:LoadAnimation(anim):Play()

-- Effetto particellare continuo
local part = workspace.EffettoPart
local emitter = Instance.new("ParticleEmitter", part)
emitter.Rate = 10
emitter.Lifetime = NumberRange.new(1,3)
